    int n,k;
    cin>>n>>k;
    vector<vector<long long>>dist(n,vector<long long>(n,1e18));
    while(k--){
        long long u,v,w;
        cin>>u>>v>>w;
        u--;
        v--;
        dist[u][v]=min(dist[u][v],w);
        dist[v][u]=min(dist[v][u],w);
    }
    for(int i=0; i<n; i++) dist[i][i]=0;

    for(int k=0; k<n; k++){
     for(int i=0; i<n; i++){
        for(int j=0; j<n; j++){
            dist[i][j]=min(dist[i][j],dist[i][k]+dist[k][j]);
        }
     }
    }
